#include <iostream>

using namespace std;
template<typename s>

void sortare(s v[], int n)
{
    char aux;
    for (int i = 0;i < n;i++) {
        for (int j = 0;j < n;j++) {
            if (v[i] != v[j])
            {
                aux = v[i];
                v[i] = v[j];
                v[j] = aux;
            }
        }
    }

    for (int i = 0; i <= n; j++) {
        cout << v[i] << ' ';
    }
}

int main()
{
    s[1001];
    int n;
    cin >> n;
    for (int i = 0;i < n;i++) {
        cin >> v[i];
    }
    sortare(v, n);
    return 0;
}
